30  get_dimensions_and_parameter_files(char **argv)
31  {
32  argh::parser cli(argv, argh::parser::PREFER_PARAM_FOR_UNREG_OPTION);
33  int dim = 2;
34  int spacedim = 2;
35  std::string input_parameter_file = "";
36  std::string output_parameter_file = "";
37 
38  const std::string exename =
39  cli(0).str().substr(cli(0).str().find_last_of("/") + 1);
40 
41 
42  // Either as -i prm_file or as first positional argument after all options
43  cli({"i", "input_prm_file"}, input_parameter_file) >> input_parameter_file;
44  cli(1, input_parameter_file) >> input_parameter_file;
45 
46  // Now the logic to deduce dim and spacedim from the parameter file name
47  bool file_contains_dim_spacedim = false;
48  if (input_parameter_file.find("1d_2d") != std::string::npos)
49  {
50  dim = 1;
51  spacedim = 2;
52  file_contains_dim_spacedim = true;
53  }
54  else if (input_parameter_file.find("1d_3d") != std::string::npos)
55  {
56  dim = 1;
57  spacedim = 3;
58  file_contains_dim_spacedim = true;
59  }
60  else if (input_parameter_file.find("2d_3d") != std::string::npos)
61  {
62  dim = 2;
63  spacedim = 3;
64  file_contains_dim_spacedim = true;
65  }
66  else if (input_parameter_file.find("1d") != std::string::npos)
67  {
68  dim = 1;
69  spacedim = 1;
70  file_contains_dim_spacedim = true;
71  }
72  else if (input_parameter_file.find("2d") != std::string::npos)
73  {
74  dim = 2;
75  spacedim = 2;
76  file_contains_dim_spacedim = true;
77  }
78  else if (input_parameter_file.find("3d") != std::string::npos)
79  {
80  dim = 3;
81  spacedim = 3;
82  file_contains_dim_spacedim = true;
83  }
84 
85  // Make sure filename and command line arguments agree
86  if (file_contains_dim_spacedim == true)
87  {
88  int cli_dim = dim;
89  int cli_spacedim = spacedim;
90  if ((cli({"d", "dim"}) >> cli_dim) ||
91  (cli({"s", "spacedim"}) >> cli_spacedim))
92  {
93  AssertThrow(
94  (dim == cli_dim) && (spacedim == cli_spacedim),
95  dealii::ExcMessage(
96  "You have specified a parameter filename that contains a "
97  "specification of the dimension and of the space dimension, "
98  "e.g., 1d_2d but you also indicated a -d or -s argument on the "
99  "command line that do not match the file name. Use only one of "
100  "the two ways to select the dimension and the space dimension, "
101  "or make sure that what you specify on the filename matches "
102  "what you specify on the command line."));
103  }
104  }
105  else
106  {
107  // get dim and spacedim from command line
108  cli({"d", "dim"}, dim) >> dim;
109  // By default spacedim is equal to dim
110  cli({"s", "spacedim"}, dim) >> spacedim;
111  }
112 
113  // Now the logic to deduce the output parameter file name. Make sure we
114  // output in the current directory, even if the file is specified with a
115  // full path
116  if (input_parameter_file != "")
117  {
118  auto rel_name = input_parameter_file.substr(
119  input_parameter_file.find_last_of("/") + 1);
120  output_parameter_file = "used_" + rel_name;
121  }
122  else if (dim == spacedim)
123  {
124  output_parameter_file =
125  "used_" + exename + "_" + std::to_string(dim) + "d.prm";
126  }
127  else
128  {
129  output_parameter_file = "used_" + exename + "_" + std::to_string(dim) +
130  "d_" + std::to_string(spacedim) + "d.prm";
131  }
132 
133  // If you want to overwrite the output parameter file, use the -o option
134  cli({"o", "output_prm_file"}, output_parameter_file) >>
135  output_parameter_file;
136 
137  deallog << "Will run in dimension " << dim << " and spacedimemsion "
138  << spacedim << std::endl
139  << "Input parameter file: " << input_parameter_file << std::endl
140  << "Output parameter file: " << output_parameter_file << std::endl;
141 
142  return std::make_tuple(dim,
143  spacedim,
144  input_parameter_file,
145  output_parameter_file);
146  }
