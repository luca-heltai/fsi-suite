31  std::tuple<int, int, std::string, std::string>
32  get_dimensions_and_parameter_files(char **argv)
33  {
34  argh::parser cli(argv, argh::parser::PREFER_PARAM_FOR_UNREG_OPTION);
35  int dim = 2;
36  int spacedim = 2;
37  std::string input_parameter_file = "";
38  std::string output_parameter_file = "";
39 
40  const std::string exename =
41  cli(0).str().substr(cli(0).str().find_last_of("/") + 1);
42 
43 
44  // Either as -i prm_file or as first positional argument after all options
45  cli({"i", "input_prm_file"}, input_parameter_file) >> input_parameter_file;
46  cli(1, input_parameter_file) >> input_parameter_file;
47 
48  // Now the logic to deduce dim and spacedim from the parameter file name
49  bool file_contains_dim_spacedim = false;
50  if (input_parameter_file.find("1d_2d") != std::string::npos)
51  {
52  dim = 1;
53  spacedim = 2;
54  file_contains_dim_spacedim = true;
55  }
56  else if (input_parameter_file.find("1d_3d") != std::string::npos)
57  {
58  dim = 1;
59  spacedim = 3;
60  file_contains_dim_spacedim = true;
61  }
62  else if (input_parameter_file.find("2d_3d") != std::string::npos)
63  {
64  dim = 2;
65  spacedim = 3;
66  file_contains_dim_spacedim = true;
67  }
68  else if (input_parameter_file.find("1d") != std::string::npos)
69  {
70  dim = 1;
71  spacedim = 1;
72  file_contains_dim_spacedim = true;
73  }
74  else if (input_parameter_file.find("2d") != std::string::npos)
75  {
76  dim = 2;
77  spacedim = 2;
78  file_contains_dim_spacedim = true;
79  }
80  else if (input_parameter_file.find("3d") != std::string::npos)
81  {
82  dim = 3;
83  spacedim = 3;
84  file_contains_dim_spacedim = true;
85  }
86 
87  // Make sure filename and command line arguments agree
88  if (file_contains_dim_spacedim == true)
89  {
90  int cli_dim = dim;
91  int cli_spacedim = spacedim;
92  if ((cli({"d", "dim"}) >> cli_dim) ||
93  (cli({"s", "spacedim"}) >> cli_spacedim))
94  {
95  AssertThrow(
96  (dim == cli_dim) && (spacedim == cli_spacedim),
97  dealii::ExcMessage(
98  "You have specified a parameter filename that contains a "
99  "specification of the dimension and of the space dimension, "
100  "e.g., 1d_2d but you also indicated a -d or -s argument on the "
101  "command line that do not match the file name. Use only one of "
102  "the two ways to select the dimension and the space dimension, "
103  "or make sure that what you specify on the filename matches "
104  "what you specify on the command line."));
105  }
106  }
107  else
108  {
109  // get dim and spacedim from command line
110  cli({"d", "dim"}, dim) >> dim;
111  // By default spacedim is equal to dim
112  cli({"s", "spacedim"}, dim) >> spacedim;
113  }
114 
115  // Now the logic to deduce the output parameter file name. Make sure we
116  // output in the current directory, even if the file is specified with a
117  // full path
118  if (input_parameter_file != "")
119  {
120  auto rel_name = input_parameter_file.substr(
121  input_parameter_file.find_last_of("/") + 1);
122  output_parameter_file = "used_" + rel_name;
123  }
124  else if (dim == spacedim)
125  {
126  output_parameter_file =
127  "used_" + exename + "_" + std::to_string(dim) + "d.prm";
128  }
129  else
130  {
131  output_parameter_file = "used_" + exename + "_" + std::to_string(dim) +
132  "d_" + std::to_string(spacedim) + "d.prm";
133  }
134 
135  // If you want to overwrite the output parameter file, use the -o option
136  cli({"o", "output_prm_file"}, output_parameter_file) >>
137  output_parameter_file;
138 
139  deallog << "Will run in dimension " << dim << " and spacedimemsion "
140  << spacedim << std::endl
141  << "Input parameter file: " << input_parameter_file << std::endl
142  << "Output parameter file: " << output_parameter_file << std::endl;
143 
144  return std::make_tuple(dim,
145  spacedim,
146  input_parameter_file,
147  output_parameter_file);
148  }
