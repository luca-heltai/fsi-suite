73  template <int dim>
74  void
75  Stokes<dim>::setup_system()
76  {
77  deallog << "System setup" << std::endl;
78  // No mixed grids
79  const auto ref_cells = triangulation.get_reference_cells();
80  AssertThrow(
81  ref_cells.size() == 1,
82  ExcMessage(
83  "This program does nots support mixed simplx/hex grid types."));
84 
85  // Compatible FE space and grid.
86  AssertThrow(finite_element().reference_cell() == ref_cells[0],
87  ExcMessage("The finite element must be defined on the same "
88  "cell type as the grid."));
89 
90  boundary_conditions.update_user_substitution_map(constants);
91  exact_solution.update_constants(constants);
92  forcing_term.update_constants(constants);
93 
94  dof_handler.distribute_dofs(finite_element);
95  auto blocks = ParsedTools::Components::block_indices(component_names,
96  component_names);
97 
98  DoFRenumbering::component_wise(this->dof_handler, blocks);
99 
100  dofs_per_block =
101  DoFTools::count_dofs_per_fe_block(this->dof_handler, blocks);
102 
103  mapping = get_default_linear_mapping(triangulation).clone();
104 
105  deallog << "Number of dofs " << dof_handler.n_dofs() << "("
106  << dofs_per_block[0] << " + " << dofs_per_block[1] << ")"
107  << std::endl;
108 
109  constraints.clear();
110  DoFTools::make_hanging_node_constraints(dof_handler, constraints);
111  boundary_conditions.check_consistency(triangulation);
112  boundary_conditions.apply_essential_boundary_conditions(*mapping,
113  dof_handler,
114  constraints);
115  constraints.close();
116 
117 
118  BlockDynamicSparsityPattern dsp(dofs_per_block, dofs_per_block);
119  DoFTools::make_sparsity_pattern(dof_handler, dsp, constraints, false);
120  sparsity_pattern.copy_from(dsp);
121  system_matrix.reinit(sparsity_pattern);
122  solution.reinit(dofs_per_block);
123  system_rhs.reinit(dofs_per_block);
124  boundary_conditions.apply_natural_boundary_conditions(
125  *mapping, dof_handler, constraints, system_matrix, system_rhs);
126  }

130  template <int dim>
131  void
132  Stokes<dim>::assemble_system()
133  {
134  deallog << "Assemble system" << std::endl;
135  const ReferenceCell cell_type = finite_element().reference_cell();
136 
137  const Quadrature<dim> quadrature_formula =
138  cell_type.get_gauss_type_quadrature<dim>(
139  finite_element().tensor_degree() + 1);
140 
141  Quadrature<dim - 1> face_quadrature_formula;
142  if constexpr (dim > 1)
143  {
144  // TODO: make sure we work also for wedges and pyramids
145  const ReferenceCell face_type = cell_type.face_reference_cell(0);
146  face_quadrature_formula =
147  face_type.get_gauss_type_quadrature<dim - 1>(
148  finite_element().tensor_degree() + 1);
149  }
150  else
151  {
152  face_quadrature_formula =
153  QGauss<dim - 1>(finite_element().tensor_degree() + 1);
154  }
155 
156  FEValues<dim, dim> fe_values(*mapping,
157  finite_element,
158  quadrature_formula,
159  update_values | update_gradients |
160  update_quadrature_points |
161  update_JxW_values);
162 
163  const unsigned int dofs_per_cell = finite_element().n_dofs_per_cell();
164  FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
165  Vector<double> cell_rhs(dofs_per_cell);
166  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
167  for (const auto &cell : dof_handler.active_cell_iterators())
168  {
169  fe_values.reinit(cell);
170  cell_matrix = 0;
171  cell_rhs = 0;
172  for (const unsigned int q_index :
173  fe_values.quadrature_point_indices())
174  {
175  for (const unsigned int i : fe_values.dof_indices())
176  {
177  const auto &eps_v = fe_values[velocity].gradient(i, q_index);
178  const auto &div_v =
179  fe_values[velocity].divergence(i, q_index);
180  const auto &q = fe_values[pressure].value(i, q_index);
181 
182  for (const unsigned int j : fe_values.dof_indices())
183  {
184  const auto &eps_u =
185  fe_values[velocity].gradient(j, q_index);
186  const auto &div_u =
187  fe_values[velocity].divergence(j, q_index);
188  const auto &p = fe_values[pressure].value(j, q_index);
189  cell_matrix(i, j) +=
190  (constants["eta"] * scalar_product(eps_v, eps_u) -
191  p * div_v - q * div_u + q * p / constants["eta"]) *
192  fe_values.JxW(q_index); // dx
193  }
194 
195  cell_rhs(i) +=
196  (fe_values.shape_value(i, q_index) * // phi_i(x_q)
197  forcing_term.value(fe_values.quadrature_point(q_index),
198  finite_element()
199  .system_to_component_index(i)
200  .first) * // f(x_q)
201  fe_values.JxW(q_index)); // dx
202  }
203  }
204  cell->get_dof_indices(local_dof_indices);
205  constraints.distribute_local_to_global(cell_matrix,
206  cell_rhs,
207  local_dof_indices,
208  system_matrix,
209  system_rhs);
210  }
211  }

315  template <int dim>
316  void
317  Stokes<dim>::run()
318  {
319  deallog.pop();
320  deallog.depth_console(console_level);
321  grid_generator.generate(triangulation);
322  for (unsigned int cycle = 0;
323  cycle < grid_refinement.get_n_refinement_cycles();
324  ++cycle)
325  {
326  deallog.push("Cycle " + Utilities::int_to_string(cycle));
327  setup_system();
328  assemble_system();
329  solve();
330  error_table.error_from_exact(dof_handler, solution, exact_solution);
331  output_results(cycle);
332  if (cycle < grid_refinement.get_n_refinement_cycles() - 1)
333  grid_refinement.estimate_mark_refine(*mapping,
334  dof_handler,
335  solution,
336  triangulation);
337  deallog.pop();
338  }
339  error_table.output_table(std::cout);
340  }
