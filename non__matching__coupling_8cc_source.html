116  NonMatchingCoupling<dim, spacedim>::assemble_dynamic_sparsity() const
117  {

119 
120  if (coupling_type == CouplingType::approximate_L2)
121  {
122  auto dsp = std::make_unique<dealii::DynamicSparsityPattern>(
123  space_dh->n_dofs(), embedded_dh->n_dofs());
124  const auto &embedded_mapping = embedded_cache->get_mapping();
125 
126  NonMatching::create_coupling_sparsity_pattern(*space_cache,
127  *space_dh,
128  *embedded_dh,
129  embedded_quadrature,
130  *dsp,
131  *space_constraints,
132  space_mask,
133  embedded_mask,
134  embedded_mapping,
135  *embedded_constraints);
136  return dsp;
137  }
138  else if (coupling_type == CouplingType::exact_L2)
139  {
140  auto dsp = std::make_unique<dealii::DynamicSparsityPattern>(
141  space_dh->n_dofs(), embedded_dh->n_dofs());
142 
143 
144  const auto &cells_and_quads =
145  NonMatching::compute_intersection(*space_cache,
146  *embedded_cache,
147  this->quadrature_order);
148  NonMatching::create_coupling_sparsity_pattern_with_exact_intersections(
149  cells_and_quads,
150  *space_dh,
151  *embedded_dh,
152  *dsp,
153  *space_constraints,
154  space_mask,
155  embedded_mask,
156  *embedded_constraints);
157  return dsp;
158  }
159  else
160  {
161  AssertThrow(
162  false, ExcMessage("The requested coupling type is not implemented."));
163  }

167  template <int dim, int spacedim>
168  void
169  NonMatchingCoupling<dim, spacedim>::adjust_grid_refinements(
170  Triangulation<spacedim, spacedim> &space_tria,
171  Triangulation<dim, spacedim> & embedded_tria,
172  const bool apply_delta_refinements) const
173  {
174  Assert(space_dh, ExcNotInitialized());
175  Assert(&embedded_tria == &embedded_dh->get_triangulation(),
176  ExcMessage(
177  "The passed embedded triangulation must be the same as the "
178  "one used by the embedded DoFHandler with which you "
179  "initialized this class."));
180  Assert(&space_tria == &space_dh->get_triangulation(),
181  ExcMessage("The passed space triangulation must be the same as the "
182  "one used by the space DoFHandler with which you "
183  "initialized this class."));
184 
185  namespace bgi = boost::geometry::index;
186 
187  auto refine = [&]() {
188  bool done = false;
189 
190  double min_embedded = 1e10;
191  double max_embedded = 0;
192  double min_space = 1e10;
193  double max_space = 0;
194 
195  while (done == false)
196  {
197  // Bounding boxes of the space grid
198  const auto &tree =
199  space_cache->get_locally_owned_cell_bounding_boxes_rtree();
200 
201  // Bounding boxes of the embedded grid
202  const auto &embedded_tree =
203  embedded_cache->get_locally_owned_cell_bounding_boxes_rtree();
204 
205  // Let's check all cells whose bounding box contains an embedded
206  // bounding box
207  done = true;
208 
209  const bool use_space =
210  ((this->refinement_strategy & RefinementStrategy::refine_space) ==
211  RefinementStrategy::refine_space);
212 
213  const bool use_embedded = ((this->refinement_strategy &
214  RefinementStrategy::refine_embedded) ==
215  RefinementStrategy::refine_embedded);
216  AssertThrow(!(use_embedded && use_space),
217  ExcMessage("You can't refine both the embedded and "
218  "the space grid at the same time."));
219 
220  for (const auto &[embedded_box, embedded_cell] : embedded_tree)
221  {
222  const auto &[p1, p2] = embedded_box.get_boundary_points();
223  const auto diameter = p1.distance(p2);
224  min_embedded = std::min(min_embedded, diameter);
225  max_embedded = std::max(max_embedded, diameter);
226 
227  for (const auto &[space_box, space_cell] :
228  tree | bgi::adaptors::queried(bgi::intersects(embedded_box)))
229  {
230  const auto &[sp1, sp2] = space_box.get_boundary_points();
231  const auto space_diameter = sp1.distance(sp2);
232  min_space = std::min(min_space, space_diameter);
233  max_space = std::max(max_space, space_diameter);
234 
235  if (use_embedded && space_diameter < diameter)
236  {
237  embedded_cell->set_refine_flag();
238  done = false;
239  }
240  if (use_space && diameter < space_diameter)
241  {
242  space_cell->set_refine_flag();
243  done = false;
244  }
245  }
246  }
247  if (done == false)
248  {
249  if (use_embedded)
250  {
251  // Compute again the embedded displacement grid
252  embedded_tria.execute_coarsening_and_refinement();
253  embedded_post_refinemnt_signal();
254  }
255  if (use_space)
256  {
257  // Compute again the embedded displacement grid
258  space_tria.execute_coarsening_and_refinement();
259  space_post_refinemnt_signal();
260  }
261  }
262  }
263  return std::make_tuple(min_space, max_space, min_embedded, max_embedded);
264  };
265 
266  // Do the refinement loop once, to make sure we satisfy our criterions
267  refine();
268 
269  // Pre refine the space grid according to the delta refinement
270  if (apply_delta_refinements && space_pre_refinement != 0)
271  for (unsigned int i = 0; i < space_pre_refinement; ++i)
272  {
273  const auto &tree =
274  space_cache->get_locally_owned_cell_bounding_boxes_rtree();
275 
276  const auto &embedded_tree =
277  embedded_cache->get_locally_owned_cell_bounding_boxes_rtree();
278 
279  for (const auto &[embedded_box, embedded_cell] : embedded_tree)
280  for (const auto &[space_box, space_cell] :
281  tree | bgi::adaptors::queried(bgi::intersects(embedded_box)))
282  space_cell->set_refine_flag();
283  space_tria.execute_coarsening_and_refinement();
284 
285  // Make sure we signal post refinement on the space grid
286  space_post_refinemnt_signal();
287 
288  // Make sure again we satisfy our criterion after the space refinement
289  refine();
290  }
291 
292  // Post refinement on embedded grid is easy
293  if (apply_delta_refinements && embedded_post_refinement != 0)
294  {
295  embedded_tria.refine_global(embedded_post_refinement);
296  embedded_post_refinemnt_signal();
297  }
298 
299  // Check once again we satisfy our criterion, and record min/max
300  const auto [sm, sM, em, eM] = refine();
301 
302  deallog << "Space local min/max diameters : " << sm << "/" << sM
303  << std::endl
304  << "Embedded space min/max diameters: " << em << "/" << eM
305  << std::endl;
306  }
