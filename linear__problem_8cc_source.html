88  template <int dim, int spacedim, class LacType>
89  void
90  LinearProblem<dim, spacedim, LacType>::setup_system()
91  {
92  TimerOutput::Scope timer_section(timer, "setup_system");
93  pcout << "System setup" << std::endl;
94  const auto ref_cells = triangulation.get_reference_cells();
95  AssertThrow(
96  ref_cells.size() == 1,
97  ExcMessage(
98  "This program does nots support mixed simplx/hex grid types."));
99 
100  // Compatible FE space and grid.
101  AssertThrow(finite_element().reference_cell() == ref_cells[0],
102  ExcMessage("The finite element must be defined on the same "
103  "cell type as the grid."));
104 
105  dof_handler.distribute_dofs(finite_element);
106 
107  // Since our code runs both for simplex grids and for hyper-cube grids, we
108  // need to make sure that we build the correct mapping for the grid. In
109  // this code we actually use a linear mapping, independently on the order
110  // of the finite element space.
111  mapping = get_default_linear_mapping(triangulation).clone();
112  pcout << "Number of dofs " << dof_handler.n_dofs() << std::endl;
113 
114  constraints.clear();
115  DoFTools::make_hanging_node_constraints(dof_handler, constraints);
116 
117  // We now check that the boundary conditions are consistent with the
118  // triangulation object, that is, we check that the boundary indicators
119  // specified in the parameter file are actually present in the
120  // triangulation, that no boundary indicator is specified twice, and that
121  // all boundary ids of the triangulation are actually covered by the
122  // parameter file configuration.
123  boundary_conditions.check_consistency(triangulation);
124 
125  // This is where we apply essential boundary conditions. The
126  // ParsedTools::BoundaryConditions class takes care of collecting boundary
127  // ids, and calling the appropriate function to apply the boundary
128  // condition on the selected boundary ids. Essential boundary conditions
129  // need to be incorporated in the constraints of the linear system, since
130  // they are part of the definition of the solution space.
131  //
132  // Natural bondary conditions, on the other hand, need access to the rhs
133  // of the problem and are not treated via constraints. We will deal with
134  // them later on.
135  boundary_conditions.apply_essential_boundary_conditions(dof_handler,
136  constraints);
137 
138  // If necessary, derived functions can add constraints to the system here.
139  add_constraints_call_back();
140  constraints.close();
141 
142  const auto blocks =
143  ParsedTools::Components::block_indices(component_names, component_names);
144  // renumber dofs in a blockwise manner.
145  DoFRenumbering::component_wise(dof_handler, blocks);
146  dofs_per_block = DoFTools::count_dofs_per_fe_block(dof_handler, blocks);
147 
148  locally_owned_dofs =
149  dof_handler.locally_owned_dofs().split_by_block(dofs_per_block);
150 
151  IndexSet non_blocked_locally_relevant_dofs;
152  DoFTools::extract_locally_relevant_dofs(dof_handler,
153  non_blocked_locally_relevant_dofs);
154  locally_relevant_dofs =
155  non_blocked_locally_relevant_dofs.split_by_block(dofs_per_block);
156 
157  pcout << "Number of degrees of freedom: " << dof_handler.n_dofs() << " ("
158  << Patterns::Tools::to_string(dofs_per_block) << ")" << std::endl;
159 
160  constraints.clear();
161  constraints.reinit(non_blocked_locally_relevant_dofs);
162  DoFTools::make_hanging_node_constraints(dof_handler, constraints);
163 
164  boundary_conditions.apply_essential_boundary_conditions(*mapping,
165  dof_handler,
166  constraints);
167  constraints.close();
168 
169  ScopedLACInitializer initializer(dofs_per_block,
170  locally_owned_dofs,
171  locally_relevant_dofs,
172  mpi_communicator);
173 
174  Table<2, DoFTools::Coupling> coupling(n_components, n_components);
175  for (unsigned int i = 0; i < n_components; ++i)
176  for (unsigned int j = 0; j < n_components; ++j)
177  coupling[i][j] = DoFTools::always;
178  initializer(system_block_sparsity, dof_handler, constraints, coupling);
179  initializer(system_block_sparsity, system_block_matrix);
180 
181  initializer(block_solution);
182  initializer(system_block_rhs);
183  initializer.ghosted(locally_relevant_block_solution);
184 
185  error_per_cell.reinit(triangulation.n_active_cells());
186 
187  boundary_conditions.apply_natural_boundary_conditions(*mapping,
188  dof_handler,
189  constraints,
190  system_block_matrix,
191  system_block_rhs);
192  // Update functions with standard constants
193  exact_solution.update_constants({});
194  forcing_term.update_constants({});
195 
196  // Now call anything else that may be needed from the user side
197  setup_system_call_back();
198  }

227  template <int dim, int spacedim, class LacType>
228  void
229  LinearProblem<dim, spacedim, LacType>::assemble_system()
230  {
231  TimerOutput::Scope timer_section(timer, "assemble_system");
232  Quadrature<dim> quadrature_formula =
233  ParsedTools::Components::get_cell_quadrature(
234  triangulation, finite_element().tensor_degree() + 1);
235 
236  Quadrature<dim - 1> face_quadrature_formula =
237  ParsedTools::Components::get_face_quadrature(
238  triangulation, finite_element().tensor_degree() + 1);
239 
240  ScratchData scratch(*mapping,
241  finite_element(),
242  quadrature_formula,
243  update_values | update_gradients |
244  update_quadrature_points | update_JxW_values,
245  face_quadrature_formula,
246  update_values | update_quadrature_points |
247  update_JxW_values);
248 
249  CopyData copy(finite_element().n_dofs_per_cell());
250 
251  auto worker = [&](const auto &cell, auto &scratch, auto &copy) {
252  assemble_system_one_cell(cell, scratch, copy);
253  };
254 
255  auto copier = [&](const auto &copy) { copy_one_cell(copy); };
256 
257  using CellFilter = FilteredIterator<
258  typename DoFHandler<dim, spacedim>::active_cell_iterator>;
259 
260  WorkStream::run(CellFilter(IteratorFilters::LocallyOwnedCell(),
261  dof_handler.begin_active()),
262  CellFilter(IteratorFilters::LocallyOwnedCell(),
263  dof_handler.end()),
264  worker,
265  copier,
266  scratch,
267  copy);
268 
269  system_block_matrix.compress(VectorOperation::add);
270  system_block_rhs.compress(VectorOperation::add);
271 
272  assemble_system_call_back();
273  }

368  template <int dim, int spacedim, class LacType>
369  void
370  LinearProblem<dim, spacedim, LacType>::run()
371  {
372  print_system_info();
373  grid_generator.generate(triangulation);
374  for (const auto &cycle : grid_refinement.get_refinement_cycles())
375  {
376  pcout << "Cycle " << cycle << std::endl;
377  setup_system();
378  assemble_system();
379  solve();
380  estimate(error_per_cell);
381  output_results(cycle);
382  if (cycle < grid_refinement.get_n_refinement_cycles() - 1)
383  {
384  mark(error_per_cell);
385  refine();
386  }
387  }
388  if (pcout.is_active())
389  error_table.output_table(std::cout);
390  }
