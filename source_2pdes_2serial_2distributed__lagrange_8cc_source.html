129  template <int dim, int spacedim>
130  void
131  DistributedLagrange<dim, spacedim>::adjust_embedded_grid(
132  const bool apply_delta_refinement)
133  {
134  namespace bgi = boost::geometry::index;
135  // Now get a vector of all cell centers of the embedded grid and refine
136  // them untill every cell is of diameter smaller than the space
137  // surrounding cells
138  std::vector<
139  std::tuple<Point<spacedim>, decltype(embedded_grid.begin_active())>>
140  centers;
141 
142  for (const auto &cell : embedded_grid.active_cell_iterators())
143  centers.emplace_back(
144  std::make_tuple(embedded_mapping().get_center(cell), cell));
145 
146  // Refine the space grid according to the delta refinement
147  if (apply_delta_refinement && delta_refinement != 0)
148  for (unsigned int i = 0; i < delta_refinement; ++i)
149  {
150  const auto &tree =
151  space_grid_tools_cache
152  ->get_locally_owned_cell_bounding_boxes_rtree();
153  for (const auto &[center, cell] : centers)
154  for (const auto &[space_box, space_cell] :
155  tree | bgi::adaptors::queried(bgi::contains(center)))
156  {
157  space_cell->set_refine_flag();
158  for (const auto face_no : space_cell->face_indices())
159  if (!space_cell->at_boundary(face_no))
160  space_cell->neighbor(face_no)->set_refine_flag();
161  }
162  space_grid.execute_coarsening_and_refinement();
163  }
164 
165  bool done = false;
166  while (done == false)
167  {
168  // Now refine the embedded grid if required
169  const auto &tree = space_grid_tools_cache
170  ->get_locally_owned_cell_bounding_boxes_rtree();
171  // Let's check all cells whose bounding box contains an embedded
172  // center
173  done = true;
174  namespace bgi = boost::geometry::index;
175  for (const auto &[center, cell] : centers)
176  {
177  const auto &[p1, p2] =
178  embedded_mapping().get_bounding_box(cell).get_boundary_points();
179  const auto diameter = p1.distance(p2);
180 
181  for (const auto &[space_box, space_cell] :
182  tree | bgi::adaptors::queried(bgi::contains(center)))
183  if (space_cell->diameter() < diameter)
184  {
185  cell->set_refine_flag();
186  done = false;
187  }
188  }
189  if (done == false)
190  {
191  // Compute again the embedded displacement grid
192  embedded_grid.execute_coarsening_and_refinement();
193  embedded_configuration_dh.distribute_dofs(
194  *embedded_configuration_fe);
195  embedded_configuration.reinit(embedded_configuration_dh.n_dofs());
196  embedded_mapping.initialize(embedded_configuration);
197 
198  // Rebuild the centers vector
199  centers.clear();
200 
201  for (const auto &cell : embedded_grid.active_cell_iterators())
202  centers.emplace_back(
203  std::make_tuple(embedded_mapping().get_center(cell), cell));
204  }
205  }
206 
207  const double embedded_space_maximal_diameter =
208  GridTools::maximal_cell_diameter(embedded_grid, embedded_mapping());
209  const double embedded_space_minimal_diameter =
210  GridTools::minimal_cell_diameter(embedded_grid, embedded_mapping());
211 
212  double space_minimal_diameter =
213  GridTools::minimal_cell_diameter(space_grid);
214  double space_maximal_diameter =
215  GridTools::maximal_cell_diameter(space_grid);
216 
217  deallog << "Space min/max diameters: " << space_minimal_diameter << "/"
218  << space_maximal_diameter << std::endl
219  << "Embedded space min/max diameters: "
220  << embedded_space_minimal_diameter << "/"
221  << embedded_space_maximal_diameter << std::endl;
222  }

227  DistributedLagrange<dim, spacedim>::setup_dofs()
228  {
229  TimerOutput::Scope timer_section(monitor, "setup_dofs");
230 
231  space_dh.distribute_dofs(*space_fe);
232  constraints.clear();
233  DoFTools::make_hanging_node_constraints(space_dh, constraints);
234  boundary_conditions.apply_essential_boundary_conditions(*space_mapping,
235  space_dh,
236  constraints);
237  constraints.close();
238 
239  DynamicSparsityPattern dsp(space_dh.n_dofs(), space_dh.n_dofs());
240  DoFTools::make_sparsity_pattern(space_dh, dsp, constraints);
241  stiffness_sparsity.copy_from(dsp);
242  stiffness_matrix.reinit(stiffness_sparsity);
243  solution.reinit(space_dh.n_dofs());
244  rhs.reinit(space_dh.n_dofs());
245  deallog << "Embedding dofs: " << space_dh.n_dofs() << std::endl;
246 
247  embedded_dh.distribute_dofs(*embedded_fe);
248  lambda.reinit(embedded_dh.n_dofs());
249  embedded_rhs.reinit(embedded_dh.n_dofs());
250  embedded_value.reinit(embedded_dh.n_dofs());
251  deallog << "Embedded dofs: " << embedded_dh.n_dofs() << std::endl;
252 
253  boundary_conditions.apply_natural_boundary_conditions(
254  *space_mapping, space_dh, constraints, stiffness_matrix, rhs);

283  void
284  DistributedLagrange<dim, spacedim>::assemble_system()
285  {
286  const auto space_quad = ParsedTools::Components::get_cell_quadrature(
287  space_grid, space_fe->tensor_degree() + 1);
288  const auto embedded_quad = ParsedTools::Components::get_cell_quadrature(
289  embedded_grid, embedded_fe->tensor_degree() + 1);
290  {
291  TimerOutput::Scope timer_section(monitor, "Assemble system");
292  // Stiffness matrix and rhs
293  MatrixTools::create_laplace_matrix(
294  space_dh,
295  space_quad,
296  stiffness_matrix,
297  forcing_term,
298  rhs,
299  static_cast<const Function<spacedim> *>(nullptr),
300  constraints);
301  }
302  {
303  TimerOutput::Scope timer_section(monitor, "Assemble coupling system");
304  NonMatching::create_coupling_mass_matrix(*space_grid_tools_cache,
305  space_dh,
306  embedded_dh,
307  embedded_quad,
308  coupling_matrix,
309  AffineConstraints<double>(),
310  ComponentMask(),
311  ComponentMask(),
312  embedded_mapping());
313 
314  // The rhs of the Lagrange multiplier as a function to plot
315  VectorTools::interpolate(embedded_mapping(),
316  embedded_dh,
317  embedded_value_function,
318  embedded_value);
319 
320  // The rhs of the Lagrange multiplier
321  VectorTools::create_right_hand_side(embedded_mapping(),
322  embedded_dh,
323  embedded_quad,
324  embedded_value_function,
325  embedded_rhs);
326  }

329  void
330  DistributedLagrange<dim, spacedim>::solve()
331  {
332  TimerOutput::Scope timer_section(monitor, "Solve system");
333 
334  auto A = linear_operator(stiffness_matrix);
335  auto Bt = linear_operator(coupling_matrix);
336  auto B = transpose_operator(Bt);
337  auto A_inv = A;
338  SparseDirectUMFPACK A_inv_direct;
339 
340  // auto control = stiffness_inverse_operator.setup_new_solver_control();
341  // TrilinosWrappers::SolverDirect solver_direct(
342  // *control,
343  // TrilinosWrappers::SolverDirect::AdditionalData(true,
344  // "Amesos_Pardiso"));
345 
346  if (use_direct_solver)
347  {
348  A_inv_direct.initialize(stiffness_matrix);
349  A_inv = linear_operator(A, A_inv_direct);
350 
351  // solver_direct.initialize(stiffness_matrix);
352  // A_inv = A;
353  // A_inv.vmult = [&](Vector<double> &dst, const Vector<double> &src) {
354  // solver_direct.solve(dst, src);
355  // };
356  }
357  else
358  {
359  stiffness_preconditioner.initialize(stiffness_matrix);
360  A_inv = stiffness_inverse_operator(A, stiffness_preconditioner);
361  }
362 
363 
364 
365  auto S = B * A_inv * Bt;
366  auto S_prec = identity_operator(S);
367  auto S_inv = schur_inverse_operator(S, S_prec);
368 
369  lambda = S_inv * (B * A_inv * rhs - embedded_rhs);
370  solution = A_inv * (rhs - Bt * lambda);
371  constraints.distribute(solution);
372  }

400  template <int dim, int spacedim>
401  void
402  DistributedLagrange<dim, spacedim>::run()
403  {
404  deallog.depth_console(console_level);
405  generate_grids_and_fes();
406  for (const auto &cycle : grid_refinement.get_refinement_cycles())
407  {
408  deallog.push("Cycle " + Utilities::int_to_string(cycle));
409  setup_dofs();
410  setup_coupling();
411  assemble_system();
412  solve();
413  error_table.error_from_exact(space_dh, solution, exact_solution);
414  output_results(cycle);
415  if (cycle < grid_refinement.get_n_refinement_cycles() - 1)
416  {
417  grid_refinement.estimate_mark_refine(space_dh,
418  solution,
419  space_grid);
420  adjust_embedded_grid(false);
421  }
422  deallog.pop();
423  }
424  error_table.output_table(std::cout);
425  }
