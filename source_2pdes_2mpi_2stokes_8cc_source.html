28  template <int dim>
29  Stokes<dim>::Stokes()
30  : LinearProblem<dim>(
31  ParsedTools::Components::join(std::vector<std::string>(dim, "u"),
32  ",") +
33  ",p",
34  "Stokes")
35  , constants("/Stokes/Constants", {"eta"}, {1.0}, {"Viscosity"})
36  , ilu_preconditioner("/Stokes/Solver/ILU preconditioner")
37  , velocity(0)
38  , pressure(dim)
39  {
40  // Fix first pressure dof to zero
41  this->add_constraints_call_back.connect([&]() {
42  // search for first pressure dof
43  unsigned int first_pressure_dof = 0;
44  for (unsigned int i = 0; i < this->finite_element().dofs_per_cell; ++i)
45  {
46  if (this->finite_element().system_to_component_index(i).first ==
47  dim)
48  {
49  first_pressure_dof = i;
50  break;
51  }
52  }
53  std::vector<types::global_dof_index> dof_indices(
54  this->finite_element().dofs_per_cell);
55  this->dof_handler.begin_active()->get_dof_indices(dof_indices);
56  this->constraints.add_line(dof_indices[first_pressure_dof]);
57  });
58  }

62  template <int dim>
63  void
64  Stokes<dim>::assemble_system_one_cell(
65  const typename DoFHandler<dim>::active_cell_iterator &cell,
66  ScratchData & scratch,
67  CopyData & copy)
68  {
69  auto &cell_matrix = copy.matrices[0];
70  auto &cell_rhs = copy.vectors[0];
71 
72  cell->get_dof_indices(copy.local_dof_indices[0]);
73 
74  const auto &fe_values = scratch.reinit(cell);
75  cell_matrix = 0;
76  cell_rhs = 0;
77 
78  for (const unsigned int q_index : fe_values.quadrature_point_indices())
79  {
80  for (const unsigned int i : fe_values.dof_indices())
81  {
82  const auto &eps_v =
83  fe_values[velocity].symmetric_gradient(i, q_index);
84  const auto &div_v = fe_values[velocity].divergence(i, q_index);
85  const auto &q = fe_values[pressure].value(i, q_index);
86 
87  for (const unsigned int j : fe_values.dof_indices())
88  {
89  const auto &eps_u =
90  fe_values[velocity].symmetric_gradient(j, q_index);
91  const auto &div_u =
92  fe_values[velocity].divergence(j, q_index);
93  const auto &p = fe_values[pressure].value(i, q_index);
94  // We assemble also the mass matrix for the pressure, to be
95  // used as a preconditioner
96  cell_matrix(i, j) +=
97  (constants["eta"] * eps_v * eps_u + div_v * p + div_u * q +
98  1 / constants["eta"] * p * q) *
99  fe_values.JxW(q_index); // dx
100  }
101 
102  cell_rhs(i) +=
103  (fe_values.shape_value(i, q_index) * // phi_i(x_q)
104  this->forcing_term.value(fe_values.quadrature_point(q_index),
105  this->finite_element()
106  .system_to_component_index(i)
107  .first) * // f(x_q)
108  fe_values.JxW(q_index)); // dx
109  }
110  }
111  }

115  template <int dim>
116  void
117  Stokes<dim>::solve()
118  {
119  TimerOutput::Scope timer_section(this->timer, "solve");
120 
121  using Vec = LA::MPI::Vector;
122  using BVec = LA::MPI::BlockVector;
123  using LinOp = LinearOperator<Vec>;
124  using BlockLinOp = BlockLinearOperator<BVec>;
125 
126  const auto &m = this->system_block_matrix;
127 
128  const auto A = linear_operator<Vec>(m.block(0, 0));
129  const auto Bt = linear_operator<Vec>(m.block(0, 1));
130  const auto B = linear_operator<Vec>(m.block(1, 0));
131  const auto Mp = linear_operator<Vec>(m.block(1, 1));
132  const auto Zero = Mp * 0.0;
133 
134  const std::array<std::array<LinOp, 2>, 2> system_ops(
135  {{{{A, Bt}}, {{B, Zero}}}});
136 
137  auto AA = block_operator<2, 2, BVec>(system_ops);
138 
139  // auto AA = block_operator<BVec>(m);
140  // AA.block(1, 1) *= 0;
141 
142 
143  this->preconditioner.initialize(m.block(0, 0));
144  ilu_preconditioner.initialize(m.block(1, 1));
145 
146  auto precA = linear_operator<Vec>(A, this->preconditioner);
147  auto precS = -1.0 * linear_operator<Vec>(Mp, ilu_preconditioner);
148 
149  std::array<LinOp, 2> diag_ops = {{precA, precS}};
150  auto diagprecAA = block_diagonal_operator<2, BVec>(diag_ops);
151 
152  deallog << "Preconditioners initialized" << std::endl;
153 
154  // If we use gmres or another non symmetric solver, use a non-symmetric
155  // preconditioner
156  if (this->inverse_operator.get_solver_name() != "cg")
157  {
158  const auto precAA = block_back_substitution(AA, diagprecAA);
159  const auto inv = this->inverse_operator(AA, precAA);
160  this->block_solution = inv * this->system_block_rhs;
161  }
162  else
163  {
164  const auto inv = this->inverse_operator(AA, diagprecAA);
165  this->block_solution = inv * this->system_block_rhs;
166  }
167 
168  this->constraints.distribute(this->block_solution);
169  this->locally_relevant_block_solution = this->block_solution;
170  }
