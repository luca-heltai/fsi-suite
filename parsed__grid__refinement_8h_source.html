307  template <int dim, int spacedim>
308  void
309  ParsedGridRefinement::limit_levels(
310  dealii::Triangulation<dim, spacedim> &tria) const
311  {
312  if (min_level != 0 || max_level != 0)
313  for (const auto cell : tria.active_cell_iterators())
314  if (cell->is_locally_owned())
315  {
316  if (min_level != 0 && cell->level() < min_level)
317  cell->set_refine_flag();
318  else if (max_level != 0 && cell->level() > max_level)
319  cell->set_coarsen_flag();
320  else if (max_level != 0 && cell->level() == max_level)
321  cell->clear_refine_flag();
322  else if (min_level != 0 && cell->level() == min_level)
323  cell->clear_coarsen_flag();
324  }
325  }
