249  template <int spacedim>
250  template <int dim, typename MatrixType, typename VectorType>
251  void
252  BoundaryConditions<spacedim>::apply_natural_boundary_conditions(
253  const dealii::Mapping<dim, spacedim> & mapping,
254  const dealii::DoFHandler<dim, spacedim> &dof_handler,
255  const dealii::AffineConstraints<double> &constraints,
256  MatrixType & matrix,
257  VectorType & rhs) const
258  {
259  for (unsigned int i = 0; i < n_boundary_conditions; ++i)
260  if (bc_type[i] == BoundaryConditionType::neumann)
261  {
262  const auto &neumann_ids = ids[i];
263  const auto &function = functions[i];
264  const auto &mask = masks[i];
265  const auto &type = types[i];
266  const auto &fe = dof_handler.get_fe();
267 
268  if (type == Components::Type::normal ||
269  type == Components::Type::tangential)
270  AssertThrow(false,
271  dealii::ExcNotImplemented(
272  "Neumann boundary conditions for normal and "
273  "tangential components are not implemented yet."));
274 
275  const auto face_quadrature_formula =
276  Components::get_face_quadrature(dof_handler.get_triangulation(),
277  fe.tensor_degree() + 1);
278 
279  dealii::FEFaceValues<dim, spacedim> fe_face_values(
280  mapping,
281  fe,
282  face_quadrature_formula,
283  dealii::update_values | dealii::update_quadrature_points |
284  dealii::update_JxW_values);
285 
286  const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
287  dealii::FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
288 
289  dealii::Vector<double> cell_rhs(dofs_per_cell);
290 
291  std::vector<dealii::types::global_dof_index> local_dof_indices(
292  dofs_per_cell);
293  for (const auto &cell : dof_handler.active_cell_iterators())
294  if (cell->at_boundary())
295  {
296  cell_rhs = 0;
297  // for(const auto face: cell->face_indices())
298  for (const unsigned int f : cell->face_indices())
299  if (neumann_ids.find(cell->face(f)->boundary_id()) !=
300  neumann_ids.end())
301  {
302  fe_face_values.reinit(cell, f);
303  for (const unsigned int i : fe_face_values.dof_indices())
304  {
305  const auto comp_i =
306  fe.system_to_component_index(i).first;
307  if (mask[comp_i])
308  for (const unsigned int q_index :
309  fe_face_values.quadrature_point_indices())
310  cell_rhs(i) +=
311  fe_face_values.shape_value(i, q_index) *
312  function->value(fe_face_values.quadrature_point(
313  q_index),
314  comp_i) *
315  fe_face_values.JxW(q_index);
316  }
317  }
318  cell->get_dof_indices(local_dof_indices);
319  constraints.distribute_local_to_global(cell_rhs,
320  local_dof_indices,
321  rhs);
322  }
323  }
324  (void)matrix;
325  }

360  template <int spacedim>
361  template <int dim>
362  void
363  BoundaryConditions<spacedim>::apply_essential_boundary_conditions(
364  const dealii::Mapping<dim, spacedim> & mapping,
365  const dealii::DoFHandler<dim, spacedim> &dof_handler,
366  dealii::AffineConstraints<double> & constraints) const
367  {
368  // Take care of boundary conditions that don't need anything else than
369  // the constraints.
370  for (unsigned int i = 0; i < n_boundary_conditions; ++i)
371  {
372  const auto &boundary_ids = ids[i];
373  const auto &bc = bc_type[i];
374  const auto &function = functions[i];
375  const auto &mask = masks[i];
376  const auto &type = types[i];
377  std::map<dealii::types::boundary_id, const dealii::Function<spacedim> *>
378  fmap;
379 
380  if (boundary_ids.find(dealii::numbers::internal_face_boundary_id) !=
381  boundary_ids.end())
382  {
383  const auto all_ids =
384  dof_handler.get_triangulation().get_boundary_ids();
385  for (const auto &id : all_ids)
386  fmap[id] = function.get();
387  }
388  else
389  for (const auto &id : boundary_ids)
390  fmap[id] = function.get();
391 
392  // In this function, we only do Dirichlet boundary conditions.
393  switch (bc)
394  {
395  case BoundaryConditionType::dirichlet:
396  switch (type)
397  {
398  case Components::Type::normal:
399  if constexpr (dim == spacedim && dim > 1)
400  dealii::VectorTools::
401  compute_nonzero_normal_flux_constraints(
402  dof_handler,
403  mask.first_selected_component(),
404  boundary_ids,
405  fmap,
406  constraints,
407  mapping);
408  else
409  AssertThrow(false,
410  dealii::ExcMessage(
411  "Cannot use normal "
412  "flux boundary conditions "
413  "for this dim and spacedim"));
414  break;
415  case Components::Type::tangential:
416  if constexpr (dim == spacedim && dim > 1)
417  dealii::VectorTools::
418  compute_nonzero_tangential_flux_constraints(
419  dof_handler,
420  mask.first_selected_component(),
421  boundary_ids,
422  fmap,
423  constraints,
424  mapping);
425  else
426  AssertThrow(false,
427  dealii::ExcMessage(
428  "Cannot use tangential "
429  "flux boundary conditions "
430  "for this dim and spacedim"));
431  break;
432  default:
433  dealii::VectorTools::interpolate_boundary_values(
434  mapping, dof_handler, fmap, constraints, mask);
435  break;
436  }
437  break;
438  default:
439  // Nothing to do in this function call
440  break;
441  }
442  }
443  }
