95  template <int spacedim>
96  void
97  BoundaryConditions<spacedim>::check_consistency() const
98  {
99  n_boundary_conditions = ids.size();
100  AssertThrow(n_boundary_conditions == bc_type.size(),
101  ExcMessage("The number of boundary ids must be equal to the "
102  "number of boundary condition types."));
103  AssertThrow(n_boundary_conditions == expressions.size(),
104  ExcMessage("The number of boundary ids must be equal to the "
105  "number of expressions."));
106  AssertThrow(n_boundary_conditions == selected_components.size(),
107  ExcMessage("The number of boundary ids "
108  "must be equal to the number "
109  "of selected components."));
110 
111  std::set<types::boundary_id> all_ids;
112  unsigned int n_ids = 0;
113  for (const auto &id : ids)
114  {
115  all_ids.insert(id.begin(), id.end());
116  n_ids += id.size();
117  }
118 
119  // Special meaning of boundary id numbers::invalid_boundary_id:
120  if (all_ids.find(numbers::invalid_boundary_id) != all_ids.end())
121  {
122  AssertThrow(all_ids.size() == 1,
123  ExcMessage("If you use the boundary id -1, then no other "
124  "boundary ids can be specified"));
125  AssertThrow(n_boundary_conditions == 1,
126  ExcMessage("Only one BoundaryCondition can be specified "
127  "with the special boundary id -1"));
128  }
129  else
130  {
131  AssertThrow(all_ids.size() == n_ids,
132  ExcMessage("You specified the same "
133  " boundary id more than once "
134  "in two different boundary conditions"));
135  // We check consistency with the triangulation
136  if (grid_info.n_active_cells > 0)
137  {
138  AssertThrow(all_ids.size() == grid_info.boundary_ids.size(),
139  ExcMessage("The number of boundary ids specified in "
140  "the input file does not match the number "
141  "of boundary ids in the triangulation"));
142  }
143  }
144 
145  // Now check that the types are valid in this dimension
146  AssertDimension(n_boundary_conditions, types.size());
147  AssertDimension(n_boundary_conditions, masks.size());
148  for (unsigned int i = 0; i < n_boundary_conditions; ++i)
149  {
150  if (types[i] == Components::Type::vector ||
151  types[i] == Components::Type::normal ||
152  types[i] == Components::Type::tangential)
153  {
154  AssertThrow(masks[i].n_selected_components() == spacedim,
155  ExcDimensionMismatch(masks[i].n_selected_components(),
156  spacedim));
157  }
158 
159  if (types[i] == Components::Type::normal ||
160  types[i] == Components::Type::tangential)
161  {
162  AssertThrow(functions[i]->n_components == spacedim,
163  ExcDimensionMismatch(functions[i]->n_components,
164  spacedim));
165  }
166  else
167  {
168  // In all other cases, it is the mask that determines the number
169  // of of components, but the function needs to be of the correct
170  // dimension
171  AssertThrow(functions[i]->n_components == n_components,
172  ExcDimensionMismatch(functions[i]->n_components,
173  n_components));
174  }
175  }
176  }
